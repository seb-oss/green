import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks'

export const Template = ({ validation, enabled, text }) => {
  const attrValidation = validation ? `className="${validation}" ` : ``
  const attrEnabled = enabled ? `` : `disabled`
  return `
    <label className="form-control">
      <input type="checkbox" ${attrValidation} ${attrEnabled} />
      <span>${text}</span>
      <i />
    </label>`
}

<Meta
  title="Components/Form/Elements/Checkbox"
  parameters={{
    componentIds: ['component-checkbox']
  }}
  argTypes={{
    validation: {
      control: 'select',
      options: ['', 'is-valid', 'is-invalid'],
    },
    enabled: {
      control: 'boolean'
    },
    text: { control: 'text' },
  }} />

<Story
  name="Checkbox"
  args={{
    text: 'Click me!',
    enabled: true
  }}
  parameters={{ docs: { disable: true }}}
>
  {Template.bind({})}
</Story>

## Group
Group form controls such as checkboxes and radio buttons using `<fieldset>` and add an optional description using a `<legend>`.
