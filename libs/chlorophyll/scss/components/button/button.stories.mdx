export const Template = ({ type, variant, text }) => {
  const attrType = type ? `type="${type}" ` : ''
  const attrClass = variant ? `class="${variant}" ` : ''
  return `<button ${attrType}${attrClass}>${text}</button>`
}

## Types

Preferably use type to describe the function of a button. Use variant only when necessary.

<Canvas>
  <button type="button">Button type</button>
  <button type="submit">Submit type</button>
  <button type="reset">Reset type</button>
</Canvas>

## Variants

Styling can also be achieved by class

<Canvas>
  <button class="primary">Primary button</button>
  <button class="secondary">Secondary button</button>
  <button class="tertiary">Tertiary button</button>
  <button class="ghost">Ghost button</button>
</Canvas>

## Sizes

Sizes can changed with size classes

<Canvas>
  <button class="primary">Default size</button>
  <button class="primary small">Small size</button>
  <button class="primary large">Large size</button>
  <button class="ghost">Default size</button>
  <button class="ghost small">Small size</button>
  <button class="ghost large">Large size</button>
</Canvas>
