  const convertAttributes = (attribs: { [key: string]: string }) => {
    const props: { [key: string]: any } = {}

    for (const [key, value] of Object.entries(attribs)) {
      if (key === 'class') {
        props.className = value
      } else if (key === 'style') {
        // Convert style string to object
        const styleObject = value
          .split(';')
          .filter((style) => style.trim())
          .reduce((acc: { [key: string]: string }, current) => {
            const [property, value] = current
              .split(':')
              .map((str) => str.trim())
            // Convert kebab-case to camelCase
            const camelProperty = property.replace(/-([a-z])/g, (g) =>
              g[1].toUpperCase(),
            )
            acc[camelProperty] = value
            return acc
          }, {})
        props.style = styleObject
      } else {
        props[key] = value
      }
    }

    return props
  }

  const getComponent = (tagName: string) => {
    // Handle icons
    if (tagName.startsWith('gds-icon-')) {
      const iconName =
        'Icon' +
        tagName
          .replace('gds-icon-', '')
          .split('-')
          .map((part: string) => part.charAt(0).toUpperCase() + part.slice(1))
          .join('')
      return (Core as any)[iconName]
    }

    // Handle regular components
    const componentName =
      'Gds' +
      tagName
        .split('-')
        .map((part: string) => part.charAt(0).toUpperCase() + part.slice(1))
        .join('')
        .replace('Gds', '')

    return (Core as any)[componentName]
  }

  const voidElements = [
    'area',
    'base',
    'br',
    'col',
    'embed',
    'hr',
    'img',
    'input',
    'link',
    'meta',
    'param',

  const parseChildren = (children: any[]): React.ReactNode => {
    // List of void elements that can't have children

    return children.map((child: any, index: number) => {
      if (child.type === 'text') {
        return child.data
      }
      if (child.type === 'tag') {
        if (child.name.startsWith('gds-')) {
          // Handle GDS components
          const Component = getComponent(child.name)
          if (Component) {
            const props = convertAttributes(child.attribs)
            return (
              <Component key={index} {...props}>
                {child.children && parseChildren(child.children)}
              </Component>
            )
          }
        } else {
          // Handle regular HTML elements
          const props = convertAttributes(child.attribs)

          // Check if it's a void element
          if (voidElements.includes(child.name)) {
            return React.createElement(child.name, { key: index, ...props })
          } else {
            return React.createElement(
              child.name,
              { key: index, ...props },
              child.children && parseChildren(child.children),
            )
          }
        }
      }
      return null
    })
  }

  const options = {
    replace: (domNode: any) => {
      try {
        if (domNode.type === 'tag') {
          if (domNode.name.startsWith('gds-')) {
            // Handle GDS components
            const Component = getComponent(domNode.name)
            if (Component) {
              const props = convertAttributes(domNode.attribs)
              return (

                  {domNode.children && parseChildren(domNode.children)}

              )
            }
          } else {
            // Handle regular HTML elements
            const props = convertAttributes(domNode.attribs)

              return React.createElement(domNode.name, props)
            } else {
              return React.createElement(
                domNode.name,
                props,
                domNode.children && parseChildren(domNode.children),
              )
            }
          }
          console.warn(`Component not found: ${domNode.name}`)
        }
      } catch (err) {
        console.warn('Error parsing component:', err)
      }
      return null

      try {
        const response = await fetch(
          `https://api.seb.io/snippets/${slug}/${slug}.json`,
        )
        if (!response.ok) {

      </Core.GdsFlex>

    console.warn('Error rendering snippet:', err)

  }
